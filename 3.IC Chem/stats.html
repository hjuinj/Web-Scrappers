# Only keeps data entries that have all the header fields defined. 
list2df <- function(list, headers){
    output = list()
    for(i in 1:length(list)){
        record = T
        for(j in headers){
            if(is.null(list[[i]][[j]])){
                record = F
                break
            }
        }
        if(record == T){
            for(j in headers)
                output[[j]] <-  c(output[[j]], list[[i]][[j]])
        }
    }
    return(as.data.frame(output))
}
